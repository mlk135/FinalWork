#include <iostream>

int main() {
    int arr[10] = { -10, -7, -4, -1, 2, 5, 8, 11, 13, 15 };

    std::cout << "Четные элементы массива: ";
    for (int i = 0; i < 10; i++) {
        if (arr[i] % 2 == 0) {  // Проверка на четность
            std::cout << arr[i] << " ";
        }
    }

    std::cout << std::endl;
    return 0;
}
